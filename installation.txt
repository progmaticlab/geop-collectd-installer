
1. завести две машины. возможные ОС:
    Ubuntu 20 LTS / Ubuntu 18 LTS / Debian 9 / Debian 10 / CentOS 7 / CentOS 8

2. установить git на обе машины
    Ubuntu | Debian: sudo apt -y install git
    CentOS:          sudo yum -y install git

3. установить ansible на обе машины
    Ubuntu | Debian: sudo apt -y install ansible
    CentOS:          sudo yum -y install ansible


на клиенте:

a. клонировать geop-collectd-installer
    git clone https://github.com/progmaticlab/geop-collectd-installer.git

b. запустить установку collectd
    cd geop-collectd-installer
    # выставить переменную telegraf_server [IP] в hosts.yaml
    sudo ansible-playbook playbooks/install-collectd.yaml -i hosts.yaml


на сервере:

a. клонировать geop-agent
    git clone https://github.com/progmaticlab/geop-agent.git

b. запустить установку telegraf
    cd geop-agent
    # выставить переменную listen_at [IP] в hosts.yaml
    sudo ansible-playbook playbooks/install-telegraf.yaml -i hosts.yaml
